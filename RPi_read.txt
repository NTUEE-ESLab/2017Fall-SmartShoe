import os
import RPi.GPIO as GPIO
GPIO.setmode(GPIO.BCM)
GPIO.setup(26,GPIO.OUT)

while True:
    text = os.popen("sudo gatttool -b A0:E6:F8:BA:20:04 --char-read -a 0x0022")..
read()
    text = text.strip("\n").split(" ")[2:6]
    text = list(map(int,text))
    for i in range(len(text)):
        text[i] -= 30
    result = text[0]*1000+text[1]*100+text[2]*10+text[3]
    print("Pressure:",result)
    if result > 500:
        GPIO.output(26,GPIO.HIGH)
    else:
        GPIO.output(26,GPIO.LOW)
